  })

  const { data, error } = await supabase.auth.signUp({
    email,
    password,
    options: {
      data: userData
    }
  })
  return { data, error }
}

  const { error } = await supabase.auth.signOut()
  return { error }
}

  const { data: { user } } = await supabase.auth.getUser()
  return user
